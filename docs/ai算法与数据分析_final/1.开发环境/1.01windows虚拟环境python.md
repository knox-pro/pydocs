### conda的简单操作
    
    
    #管理环境
    conda env list
    #创建虚拟环境
    conda create --name py36_yangtuo python=3.6 #创建一个python3.6的虚拟环境
    conda create --name py39_yangtuo python=3.9 #创建一个python3.9的虚拟环境
    #激活虚拟环境
    conda activate py36_yangtuo #新版的conda要省略前面的conda
    #退出虚拟环境
    conda deactivate
    #删除虚拟环境
    conda remove py36_yangtuo
