# 7\. 人脸案例

## 1 基础

我们使⽤机器学习的⽅法完成⼈脸检测，⾸先需要⼤量的正样本图像（⾯部图像）

和负样本图像（不含⾯部的图像）来训练分类器。我们需要从其中提取特征。下图

中的 Haar 特征会被使⽤，就像我们的卷积核，每⼀个特征是⼀ 个值，这个值等于

⿊⾊矩形中的像素值之后减去⽩⾊矩形中的像素值之和。

![](http://cyc.chenyangstudy.cn/book_chenyang/a04opencv/images/203_1.jpg)

Haar特征值反映了图像的灰度变化情况。例如：脸部的⼀些特征能由矩形特征简单

的描述，眼睛要⽐脸颊颜⾊要深，⿐梁两侧⽐⿐梁颜⾊要深，嘴巴⽐周围颜⾊要深

等。

Haar特征可⽤于于图像任意位置，⼤⼩也可以任意改变，所以矩形特征值是矩形模

版类别、矩形位置和矩形⼤⼩这三个因素的函数。故类别、⼤⼩和位置的变化，使

得很⼩的检测窗⼝含有⾮常多的矩形特征。

![](http://cyc.chenyangstudy.cn/book_chenyang/a04opencv/images/204_1.jpg)

得到图像的特征后，训练⼀个决策树构建的adaboost级联决策器来识别是否为⼈脸。

![](http://cyc.chenyangstudy.cn/book_chenyang/a04opencv/images/204_2.jpg)

那我们就利⽤这些⽂件来识别⼈脸，眼睛等。检测流程如下：

  1. 读取图⽚，并转换成灰度图

  2. 实例化⼈脸和眼睛检测的分类器对象


    
    
    # 实例化级联分类器
    
    classifier =cv.CascadeClassifier( "haarcascade_frontalface_default.xml" )
    
    # 加载分类器
    
    classifier.load('haarcascade_frontalface_default.xml')
    

  1. 进⾏⼈脸和眼睛的检测 
         
         rect = classifier.detectMultiScale(gray, scaleFactor, minNeighbors, minSiz
         


参数：

Gray: 要进⾏检测的⼈脸图像

scaleFactor: 前后两次扫描中，搜索窗⼝的⽐例系数

minneighbors：⽬标⾄少被检测到minNeighbors次才会被认为是⽬标

minsize和maxsize: ⽬标的最⼩尺⼨和最⼤尺⼨

  1. 将检测结果绘制出来就可以了


主程序如下所示：

    import cv2 as cv
    import numpy as np
    import os
    import hmmlearn.hmm as hl
    import warnings
    warnings.filterwarnings('ignore',category=DeprecationWarning)
    np.seterr(all="ignore")
    
    #1、提取图片路径
    def search_objects(directory):
        directory = os.path.normpath(directory)
        if not os.path.isdir(directory):
            raise IOError(f"您输入的路径：{directory}不存在！")
        objects = {}
        for curdir,subdir,files in os.walk(directory):
            for jpeg in (file for file in files if file.endswith(".jpg") ):
                path = os.path.join(curdir,jpeg)
                label = path.split("\\")[-2]
                if label not in objects:
                    objects[label] = []
                objects[label].append(path)
        return objects
    # print(search_objects("./objects/training"))
    #2、通过路径构造数据
    train_objects = search_objects("./objects/training")
    train_x,train_y = [],[]
    for label,filenames in train_objects.items():
        descs = np.array([])
        for filename in filenames:
            image = cv.imread(filename)
            gray = cv.cvtColor(image,cv.COLOR_BGR2GRAY)
            h,w = gray.shape[:2]
            f = 200/min(h,w)
            gray = cv.resize(gray,None,fx=f,fy=f)
            star = cv.xfeatures2d.StarDetector_create()
            keypoints = star.detect(gray)
            sift = cv.xfeatures2d.SIFT_create()
            _,desc = sift.compute(gray,keypoints)
            if len(descs) == 0:
                descs = desc
            else:
                descs = np.append(descs,desc,axis=0)
        train_x.append(descs)
        train_y.append(label)
    
    #3、构造模型，模型训练
    models = {}
    for descs,label in zip(train_x,train_y):
        model = hl.GaussianHMM(n_components=4,n_iter=1000)
        models[label] = model.fit(descs)
    
    #4、模型预测
    #构造测试数据
    test_objects = search_objects("./objects/testing")
    test_x,test_y = [],[]
    test_z = []  #可视化
    for label, filenames in test_objects.items():
        test_z.append([])
        descs = np.array([])
        for filename in filenames:
            image = cv.imread(filename)
            test_z[-1].append(image)
            gray = cv.cvtColor(image, cv.COLOR_BGR2GRAY)
            h, w = gray.shape[:2]
            f = 200 / min(h, w)
            gray = cv.resize(gray, None, fx=f, fy=f)
            star = cv.xfeatures2d.StarDetector_create()
            keypoints = star.detect(gray)
            sift = cv.xfeatures2d.SIFT_create()
            _, desc = sift.compute(gray, keypoints)
            if len(descs) == 0:
                descs = desc
            else:
                descs = np.append(descs,desc,axis=0)
        test_x.append(descs)
        test_y.append(label)
    
    #预测
    pred_test_y = []
    for descs in test_x:
        best_score,best_label = None,None
        for label,model in models.items():
            score = model.score(descs)
            if (best_score is None) or (best_score<score):
                best_score = score
                best_label = label
    
    
        pred_test_y.append(best_label)
    
    # print(pred_test_y==test_y)
    # print(pred_test_y)
    # print(test_y)
    #5、可视化
    i = 0
    for true_label,pred_label,images in zip(test_y,pred_test_y,test_z):
        for image in images:
            i+=1
            cv.imshow(f"{i}-{true_label}{'==' if true_label==pred_label else '!='}{pred_label}",image)
    
    cv.waitKey()
    

### 明星识别
    
    
    import os
    import numpy as np
    import cv2 as cv
    import sklearn.preprocessing as sp
    
    fd = cv.CascadeClassifier('./haar/face.xml')
    
    def search_faces(directory):
        directory = os.path.normpath(directory)
        if not os.path.isdir(directory):
            raise IOError(f"路径有误：{directory}")
        faces = {}
        for curdir,subdir,files in os.walk(directory):
            for jpegs in (file for file in files if file.endswith(".jpg") or file.endswith(".png")):
                path = os.path.join(curdir,jpegs)
                label = path.split(os.path.sep)[-2]
    
                if label not in faces:
                    faces[label] = []
                faces[label].append(path)
        return faces
    
    #获取数据
    train_x,train_y = [],[]
    train_faces = search_faces("./faces/training") #字典，键为标签，值为路径
    #对标签进行编码化,
    codec = sp.LabelEncoder()
    codec.fit(list(train_faces.keys()))
    for label,filenames in train_faces.items():
        for file in filenames:
            frame = cv.imread(file)
            # 灰化处理
            gray = cv.cvtColor(frame, cv.COLOR_BGR2GRAY)
            # 识别脸部,返回脸部的坐标
            faces = fd.detectMultiScale(gray, 1.3, 5)
            #将脸部信息加载到训练集,如果没有识别到脸部则不存储到数据集
            for l, t, w, h in faces:
                face = gray[t:t + h, l:l + w]
                train_x.append(face)
                train_y.append(int(codec.transform([label])[0]))
    train_y = np.array(train_y)
    # print(train_y)
    #局部二值模式直方图识别检测器模型
    model= cv.face.LBPHFaceRecognizer_create()
    #模型训练
    model.train(train_x,train_y)
    #模型预测
    test_x,test_y,test_z = [],[],[]
    test_faces = search_faces("./faces/training") #字典，键为标签，值为路径
    for label,filenames in test_faces.items():
        for file in filenames:
            frame = cv.imread(file)
    
            # 灰化处理
            gray = cv.cvtColor(frame, cv.COLOR_BGR2GRAY)
            # 识别脸部,返回脸部的坐标
            faces = fd.detectMultiScale(gray, 1.3, 5)
            #将脸部信息加载到训练集,如果没有识别到脸部则不存储到数据集
            for l, t, w, h in faces:
                face = gray[t:t + h, l:l + w]
                test_x.append(face)
                test_y.append(int(codec.transform([label])[0]))
    
                #标记脸部，再保存图片
                a = int(w / 2)
                b = int(h / 2)
                cv.ellipse(frame, (l + a, t + b), (a, b), 0, 0, 360, (255, 0, 255), 5)
                test_z.append(frame)
    test_y = np.array(test_y)
    
    pred_test_y = []
    for face in test_x:
        pred_code = model.predict(face)[0]
        pred_test_y.append(pred_code)
    
    #可视化
    while True:
        for code,pred_code,image in zip(test_y,pred_test_y,test_z):
            label = codec.inverse_transform([code])[0] #反向解码
            pred_label = codec.inverse_transform([pred_code])[0]
            text = f"{label}{'==' if label==pred_label else '!='}{pred_label}"
            cv.putText(image,text,(10,60),cv.FONT_HERSHEY_COMPLEX,2,(255,0,255),6)
            cv.imshow("Face Recongnizing",image)
    
    
            if cv.waitKey(1000) == 27:
                break
