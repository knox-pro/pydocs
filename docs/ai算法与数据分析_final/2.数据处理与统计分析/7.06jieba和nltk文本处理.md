### jieba分词模式
    
    
    import jieba
    s1 = "南京市长江大桥"
    
    
    #1、精确模式（默认）:符合人的阅读方式
    print("精确模式（默认）".center(50,"*"))
    print('/'.join(jieba.cut(s1,cut_all=False)))
    #2、全模式;返回所有可能的词
    print("全模式".center(50,"*"))
    print('/'.join(jieba.cut(s1,cut_all=True)))
    
    #3、搜索引擎模式:查找可能为关键词的所有词
    print("搜索引擎模式".center(50,"*"))
    print('/'.join(jieba.cut_for_search(s1)))
    

jieba 支持三种分词模式： * 精确模式，试图将句子最精确地切开，适合文本分析； * 全模式，把句子中所有的可以成词的词语都扫描出来, 速度非常快，但是不能解决歧 义； * 搜索引擎模式，在精确模式的基础上，对长词再次切分，提高召回率，适合用于搜 索引擎分词。

    import nltk
    from nltk.stem.porter import PorterStemmer
    from nltk.stem import WordNetLemmatizer
    from nltk.corpus import stopwords
    
    
    def nlp_english(text):
    #第一步：读取原始文本
    text = text
    #第二步:分词
    words = nltk.word_tokenize(text) #返回一个列表
    #第三步：词干提取
    p = PorterStemmer() #词干提取器
    words_list1 = [p.stem(i) for i in words]
    
        #第四步:词性还原
        w = WordNetLemmatizer()
        word_list2 = [w.lemmatize(i) for i in words_list1]
    
        #第五步;去停用词
        words = [i for i in word_list2 if i not in stopwords.words() and i != '...']
    
        return words #返回去除停用词后的单词列表

        """

        2、处理薪资，由于薪资都是一个区间，所以增加一列保存最高与最低的平均值，做为本岗位的平均薪资作为统计标准。

        data_pd = pd.read_csv("python.csv")
        print(data_pd.columns)
        #去除空值的行
        data_pd = data_pd.dropna(subset=['positionName', 'salary'],how='any')

        data_pd['city'] = data_pd.loc[:,"city"].apply(lambda i:i.split(".")[0])

        # print(data_pd["area"])
    
        return data_pd
