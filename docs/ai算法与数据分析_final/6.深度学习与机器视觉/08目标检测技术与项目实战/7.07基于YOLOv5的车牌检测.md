### 数据集标签的txt生成 （getlabeltotxt.py）

![img_78.png](http://cyc.chenyangstudy.cn/book_chenyang/a05深度学习与机器视觉/08目标检测技术与项目实战/img_78.png)
    
    
    #coding:utf8
    import shutil
    import cv2
    import os
    
    def getlabel(imagedir, txtdir):
        for filename in os.listdir(imagedir):
            ## 0451171875-94_265-224&409_554&547-551&547_225&497_224&409_554&438-0_0_3_24_32_31_30_24-148-63.jpg
            print('filename',filename)
            list1 = filename.split("-", 3)  # 第一次分割，以减号'-'做三次分割，得到车牌区域比例，角度，两个标注点信息
            # list1 = ['0451171875','94_265','224&409_554&547','551&547_225&497_224&409_554&438-0_0_3_24_32_31_30_24-148-63']
            subname = list1[2] # 两个标注点信息
            # subname ='224&409_554&547'
            lt, rb = subname.split("_")  # 第二次分割，以下划线'_'做分割, lt存储左上角坐标，rb存储剩余信息
            # lt = '224&409',rb = '554&547'
            lx, ly = lt.split("&") # 左上角坐标
            rx, ry = rb.split("&") # 右下角坐标
            width = int(rx) - int(lx)
            height = int(ry) - int(ly)  # bounding box的宽和高
            cx = float(lx) + width / 2
            cy = float(ly) + height / 2  # bounding box中心点
    
            img = cv2.imread(os.path.join(imagedir,filename))
            if img is None:
                print('processing failed',filename)
                continue
    
            # 宽高与中心点进行标准化
            width = width / img.shape[1]
            height = height / img.shape[0]
            cx = cx / img.shape[1]
            cy = cy / img.shape[0]
    
            txtname = filename.split(".", 1)
            txtfile = os.path.join(txtdir, txtname[0] + ".txt")
            # 绿牌是第0类，蓝牌是第1类
            with open(txtfile, "w") as f:
                f.write(str(0) + " " + str(cx) + " " + str(cy) + " " + str(width) + " " + str(height))
    
    
    if __name__ == '__main__':
        # 图片存储地址
        trainDir = "CCPD2020/detection/images/train"
        valDir = "CCPD2020/detection/images/val"
        testDir = "CCPD2020/detection/images/test"
        # 标签文件存储地址
        train_txtdir = "CCPD2020/detection/labels/train"
        val_txtdir = "CCPD2020/detection/labels/val"
        test_txtdir = "CCPD2020/detection/labels/test"
        if not os.path.exists(train_txtdir):
            os.mkdir(train_txtdir)
        if not os.path.exists(val_txtdir):
            os.mkdir(val_txtdir)
        if not os.path.exists(test_txtdir):
            os.mkdir(test_txtdir)
        #getlabel(trainDir, train_txtdir)
        getlabel(valDir, val_txtdir)
        #getlabel(testDir, test_txtdir)
