## PyTorch处理图片
    
    
    import torch
    from torchvision import transforms
    from PIL import Image
    
    image_path = r"image.png"
    save_path = r"result.png"
    # 加载图片
    image = Image.open(image_path)
    
    # 定义转换流程：PIL图像 -> Tensor
    transform = transforms.ToTensor()
    
    # 应用转换
    tensor_image = transform(image)
    
    print(tensor_image)
    # 检查CUDA是否可用并将tensor移至CUDA
    if torch.cuda.is_available():
        tensor_image = tensor_image.to('cuda')
    
    # 对每个元素加一
    tensor_image += 0.1
    
    # 将tensor移回CPU并转换回PIL图像
    tensor_image = tensor_image.to('cpu')
    transformed_image = transforms.ToPILImage()(tensor_image)
    
    # 保存图像
    transformed_image.save(save_path)

            return len(...)
        def __getitem__(self, idx): 
            #当数据集被读取时，需要返回的数据
            ...
            return x[idx], y[idx]
    

  * DataLoader格式说明


dataloader用来定义数据集中的数据是怎样被加载的，例如每次加载多少个数据（Batch Size），用多少个线程加载数据（Num

Workers），加载数据时是否对数据进行洗牌打乱操作（Shuffle）等，模板像下面这样： 
    
    
    my_dataloader = DataLoader(my_dataset, batch_size=2, 
                               shuffle=True, num_workers=4)
    

## 案例1：导入两个列表到Dataset
    
    
    from torch.utils.data import Dataset, DataLoader
    
    class MyDataset(Dataset):   # 继承自Dataset
        def __init__(self, ):   # 定义数据集包含了什么东西
            self.x = [i for i in range(10)]
            self.y = [2*i for i in range(10)]
        def __len__(self):      # 返回数据集的总长度
            return len(self.x)
        def __getitem__(self, idx): #当数据集被读取时，需要返回的数据
            return self.x[idx], self.y[idx]
    
    my_dataset = MyDataset()
    my_dataloader = DataLoader(my_dataset)
    
    for x_i, y_i in my_dataloader:
        print("x_i = ",x_i, " y_i = ",y_i)

    my_dataloader = DataLoader(my_dataset, batch_size=2, shuffle=True, num_workers=4)

    from torch.utils.data import Dataset, DataLoader
    import pandas as pd
    class MyDataset(Dataset):   # 继承自Dataset
        def __init__(self, data_loc):   # 定义数据集包含了什么东西
            data = pd.read_excel(data_loc)
            self.x1, self.x2, self.x3, self.x4, self.y = data['x1'], data['x2'], data['x3'], data['x4'], data['y']
    
        def __len__(self):      # 返回数据集的总长度

        def __getitem__(self, idx): #当数据集被读取时，需要返回的数据
            return self.x1[idx],self.x2[idx],self.x3[idx],self.x4[idx], self.y[idx]
    
    if __name__ == '__main__':

## 案例3：导入图像数据集到Dataset
    
    
    import os
    import cv2 as cv
    import torch
    from torch.utils.data import DataLoader, Dataset
    import numpy as np
    class MyImageDataset(Dataset):
        def __init__(self):
            image_root = r"datasets/animals"
            self.file_path_list = []
            for root, dirs, files in os.walk(image_root):
                for file_i in files:
                    file_i_full_path = os.path.join(root, file_i)
                    self.file_path_list.append(file_i_full_path)
        def __len__(self):
            return len(self.file_path_list)
    
        def __getitem__(self, item):
            img = cv.imread(self.file_path_list[item])
            img = cv.resize(img, dsize=(256,256))
            # HWC
            # 012
            # CHW
            # 210
            img = np.transpose(img, (2,1,0))
            img_tensor = torch.from_numpy(img)
            return img_tensor
    if __name__ == '__main__':
        my_image_dataset = MyImageDataset()
        my_dataloader = DataLoader(my_image_dataset)
        for x_i in my_dataloader:
            print(x_i.shape)

让返回值同时带有图片和标签 
    
    
    import os
    import cv2 as cv
    import torch
    from torch.utils.data import DataLoader, Dataset
    import numpy as np
    class MyImageDataset(Dataset):
        def __init__(self):
            image_root = r"datasets/animals"
            self.file_path_list = []
            dir_name = []
            self.labels = []
            for root, dirs, files in os.walk(image_root):
                if dirs:
                    dir_name = dirs
                for file_i in files:
                    file_i_full_path = os.path.join(root, file_i)
                    self.file_path_list.append(file_i_full_path)
    
                    label = root.split(os.sep)[-1]
                    label_id = dir_name.index(label)
                    self.labels.append(label_id)
    
        def __len__(self):
            return len(self.file_path_list)
    
        def __getitem__(self, item):
            img = cv.imread(self.file_path_list[item])
            img = cv.resize(img, dsize=(256,256))
            # HWC
            # 012
            # CHW
            # 210
            img = np.transpose(img, (2,1,0))
            img_tensor = torch.from_numpy(img)
            label = self.labels[item]
            return img_tensor,label
    if __name__ == '__main__':
        my_image_dataset = MyImageDataset()
        my_dataloader = DataLoader(my_image_dataset, batch_size=4, shuffle=True)
        for x_i, y_i in my_dataloader:
            print(x_i.shape, y_i)

## 案例4：加载官方数据集

有一些数据集是PyTorch自带的，它被保存在`TorchVision`中，详见 Datasets — Torchvision 0.15 documentation (pytorch.org)

以`mnist`数据集为例进行加载 
    
    
    import torch
    import torchvision
    
    transform = torchvision.transforms.ToTensor()
    
    trainset = torchvision.datasets.MNIST(root='./data', train=True,  download=True, transform=transform)
    trainloader = torch.utils.data.DataLoader(trainset, batch_size=4, shuffle=True)
    
    testset = torchvision.datasets.MNIST(root='./data', train=False,  download=True, transform=transform)
    testloader = torch.utils.data.DataLoader(testset, batch_size=4, shuffle=False)
    for x,y in trainloader:
        print(x.shape, y)
    
    for x,y in testloader:
        print(x.shape, y)
